interface ProcessBase {
	protected process() {
	}
}

public static enum TransactionDirection {
	FROM,
	TO,
}
	
protected abstract class TransactionDirectionBase {
	private TransactionDirection keyDirection ;
	protected processTransactionDirection() {
	}
}
public class FromImpl extends TransactionDirectionBase {
	private processTransactionDirection() {
		keyDirection = TransactionDirection.FROM ;
		// impl
	}
}
public class ToImpl extends TransactionDirectionBase {
	private processTransactionDirection() {
		keyDirection = TransactionDirection.TO ;
		// impl
	}
}
--------------
protected abstract class FromToTypeBase {
	private FromToType key ;
	protected processFromToType() {
	}
}
public class AllImpl extends FromToTypeBase {
	public processFromToType() {
		key = FTType.FromToType.All;
		// All impl
	}
}
public class RemImpl extends FromToTypeBase {
	public processFromToType() {
		key = FTType.FromToType.Remainder;
		// Rem impl
	}
}
public class IndivImpl extends FromToTypeBase {
	public processFromToType() {
		key = FTType.FromToType.Individual;
		// Indiv impl
	}
}

class FromToTypeImpl {
	AllImpl		aImpl ;
	RemImpl		rImpl ;
	IndivImpl 	iImpl ;
}

class Transaction {
	TransactionDirectionImpl	tdImpl;
	FromToTypeImpl				ftImpl ;

	
	void Transaction(TransactionDirection){
	}

	process() {
		tdImpl.ftImpl.aImpl.process(this.keyDirection) ;
		tdImpl.ftImpl.rImpl.process(this.keyDirection) ;
		tdImpl.ftImpl.iImpl.process(this.keyDirection) ;
	}
}
